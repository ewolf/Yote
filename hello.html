Yote Hello World Example

Server Side Code

package GServ::Hello;

use strict;

use base 'GServ::AppRoot';

sub hello {
    my( $self, $data, $acct ) = @_;
    my $name = $data->{name};
    return { r => "hello there '$name'" };
}

1;
